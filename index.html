// --- Configuration & Persistence --- // This lets you change your proxy URL via the browser console or a prompt // without editing the HTML file every time. let PROXY_BASE_URL = localStorage.getItem('proxy_url') || 'https://api.codetabs.com/v1/proxy?quest='; // --- Proxy & Smart Search Logic --- const proxyForm = document.getElementById('proxyForm'); const proxyInput = document.getElementById('proxyInput'); proxyForm.addEventListener('submit', function(e) { e.preventDefault(); let input = proxyInput.value.trim(); if (!input) return; let finalUrl = ""; // Check if the input looks like a URL (contains a dot and no spaces) const isUrl = input.includes('.') && !input.includes(' '); if (isUrl) { // Ensure protocol exists if (!input.startsWith('http://') && !input.startsWith('https://')) { input = 'https://' + input; } finalUrl = PROXY_BASE_URL + encodeURIComponent(input); } else { // It's a search term, send to Google finalUrl = `https://www.google.com/search?q=${encodeURIComponent(input)}`; } window.open(finalUrl, '_blank'); proxyInput.value = ''; }); // --- Audio Player Logic --- const fileInput = document.getElementById('fileInput'); const audioPlayer = document.getElementById('audioPlayer'); fileInput.addEventListener('change', function() { const file = this.files[0]; if (file) { const fileURL = URL.createObjectURL(file); audioPlayer.src = fileURL; audioPlayer.play(); // Optional: Update title to file name document.querySelector('h1').textContent = "Now Playing: " + file.name.substring(0, 15) + "..."; } }); // --- Live Clock Logic --- function updateClock() { const now = new Date(); document.getElementById('clock').textContent = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit', second: '2-digit' }); } setInterval(updateClock, 1000); updateClock();
